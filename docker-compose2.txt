version: "3.9"

  nginx:
    image: nginx:latest
    container_name: nginx
    ports:
      - "80:80"
    networks:
      - app-network

  frontend:
    build: ./frontend
    container_name: frontend
    networks:
      - app-network

  backend:

    networks:
      - app-network

    container_name: mysql
    environment:

    build: ./admin
    container_name: admin
    networks:
      - app-network

  metadata:
    build: ./metadata
    container_name: metadata
    networks:
      - app-network
